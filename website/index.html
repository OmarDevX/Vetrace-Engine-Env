Readable game code

Spawn an actor. Add behavior. Keep moving.

Vetrace keeps common gameplay operations direct while preserving strict engine boundaries underneath.

Learn the Actor API →
src/main.rs
use glam::{Quat, Vec3};
use vetrace_core::{Engine, Transform};
use vetrace_render::{
    Material, PrimitiveShape, Renderable, Shape,
};

let mut engine = Engine::new();

let player = engine
    .spawn_actor("Player")
    .with(Transform {
        translation: Vec3::Y,
        rotation: Quat::IDENTITY,
        scale: Vec3::ONE,
    })
    .with(Shape {
        primitive: PrimitiveShape::Cube,
        size: Vec3::ONE,
    })
    .with(Material::default())
    .with(Renderable {
        mesh: None,
        material: None,
        visible: true,
    })
    .tag("player")
    .build();
